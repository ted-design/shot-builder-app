Implementation CSS

/* Image container - fixed height, flexible width */
.image-container {
  height: 120px;              /* or max-height for more flexibility */
  display: flex;
  align-items: center;        /* Vertical centering */
  justify-content: center;    /* Horizontal centering */
  background: transparent;    /* Invisible letterboxing */
}

/* Image itself - fits within bounds, maintains aspect ratio */
.image-container img {
  max-width: 100%;
  max-height: 100%;
  object-fit: contain;        /* Key: show full image, no crop */
  border-radius: 6px;         /* Rounded corners on image, not container */
}

/* Fallback for no-image state */
.image-container .placeholder {
  width: 100%;
  height: 100%;
  display: flex;
  align-items: center;
  justify-content: center;
  background: rgb(241 245 249); /* slate-100 */
  border-radius: 6px;
}
